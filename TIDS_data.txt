#!/bin/bash

# UM

basedir=/group_workspaces/jasmin2/tids

outfile=TIDS_data.txt

echo \#\#\#\#\# > ~/$outfile

for i in $(seq 1 3)

    do 

    if [ $i -eq 1 ]; then

        dir=$basedir/UM/ancil/atmos/n96e

        echo $i 
        echo $dir

    fi

    if [ $i -eq 2 ]; then

        dir=$basedir/UM/ancil/atmos/n216e

        echo $i 
        echo $dir

    fi


    if [ $i -eq 3 ]; then

        dir=$basedir/OCEAN

        echo $i 
        echo $dir

    fi

    echo \#\#\#\#\#>> ~/TIDS_data.txt
    echo \#$dir >> ~/TIDS_data.txt
    echo \#\#\#\#\#>> ~/TIDS_data.txt

    find $dir -name "[!.]*" -type f >> ~/$outfile 

    paste $outfile $outfile > $outfile

done
